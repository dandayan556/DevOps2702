pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
             
               'git 'https://github.com/dandayan556/WorldOfGames
            }
        }

        stage('Build') {
            steps {
               
                sh 'docker build -t your-image-name .'
            }
        }

        stage('Run') {
            steps {
                
                sh 'docker run -d -p 8777:8777 -v 'C:\Users\mgidy\PycharmProjects\pythonProject\DevOps2702\Scores.txt' myflaskapp'
            }
        }

        stage('Test') {
            steps {
               
                sh 'python e2e.py'
            }
        }

        stage('Finalize') {
            steps {
                
                sh 'docker stop $(docker ps -q --filter ancestor=your-image-name)'
                
                sh 'docker push your-image-name'
            }
        }
    }

    post {
        always {
            
            sh 'docker rm $(docker ps -a -q)'
            sh 'docker rmi your-image-name'
        }
    }
}
